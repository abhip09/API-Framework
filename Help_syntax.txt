
let twoDarray = [];
let rows = pm.info.iterationCount;
let columns = 7;

// creating two-dimensional array
for (let i = 0; i < rows; i++) {
  twoDarray[i] = [];
  for (let j = 0; j < columns; j++) {
    twoDarray[i][j] = j;
  }
}

console.log(twoDarray);

pm.collectionVariables.set("twoDarray", twoDarray)


var Flag = pm.iterationData.get("Flag")

if (Flag === "TestData"){
    console.log("skipping this request")
    pm.execution.skipRequest()
}
